
inputs: price(numeric), period(numeric), vfactor(numeric);
vars  : length(0), EP(0), GD(0), X1(0), X2(0);

length = (period + 1.0) * 0.5;
EP = 1.0 + vfactor;

X1 = ema(price, length) * EP;
X2 = ema(ema(price, length), length) * vfactor;
GD = X1 - X2;

X1 = ema(GD, length) * EP;
X2 = ema(ema(GD, length), length) * vfactor;
GD = X1 - X2;

X1 = ema(GD, length) * EP;
X2 = ema(ema(GD, length), length) * vfactor;
GD = X1 - X2;

_T3 = GD;
